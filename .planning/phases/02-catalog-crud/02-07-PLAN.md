```typescript
import { z } from "zod";

const optionalString = z
  .string()
  .optional()
  .transform((value) => {
    const trimmed = value?.trim();
    return trimmed && trimmed.length > 0 ? trimmed : null;
  });

export const institutionalLeadSchema = z.object({
  organization: z.string().min(2, "Informe o nome da escola ou rede."),
  contactName: z.string().min(2, "Informe o nome do contato."),
  email: z.string().email("Informe um e-mail válido."),
  phone: optionalString,
  headcount: optionalString
    .transform((value) => (value ? Number(value) : null))
    .refine((value) => value === null || Number.isFinite(value), "Informe apenas números."),
  message: optionalString,
});

export type InstitutionalLeadInput = z.infer<typeof institutionalLeadSchema>;
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser URL → RSC page.tsx (searchParams) | utm_* values are untrusted query parameters; validated by Zod schema at action layer (max 255, optional) before DB insert |
| InstitutionalLeadForm (client) → submitInstitutionalLead (server action) | UTM fields arrive as FormData strings; Zod schema trims and nullifies empty values before insert |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-01 | Tampering | utm_* inputs in InstitutionalLeadForm | mitigate | UTM values flow through Zod schema in the server action: `z.string().max(255).optional().transform(...)`. Values over 255 chars are rejected before DB insert. Empty/whitespace transforms to null. SQL injection is not possible (parameterized queries via Supabase client). |
| T-07-02 | Information Disclosure | Hidden inputs visible in DOM | accept | utm_* values in hidden inputs are visible in browser DevTools. Accepted: UTM parameters are marketing metadata, not secrets. They were in the URL already; including them in DOM is not a disclosure escalation. |
| T-07-03 | Denial of Service | Overlong UTM string from URL | mitigate | Zod max(255) validates at action layer. Even without schema, Supabase TEXT column accepts any length but the `utm_*_length` CHECK constraints added in migration 0014 (max 255 enforced at DB level as well — defense in depth). |
</threat_model>
